        { "imports": { "three": "https://cdn.jsdelivr.net/npm/three@0.160.0/build/three.module.js", "three/addons/": "https://cdn.jsdelivr.net/npm/three@0.160.0/examples/jsm/" } }

    <script type="module">
        import * as THREE from 'three';
        import { OrbitControls } from 'three/addons/controls/OrbitControls.js';
        import { STLLoader } from 'three/addons/loaders/STLLoader.js';

        // --- Configuration ---
        const CONFIG = {
            particleCount: 40000,
            particleSize: 0.8,
            color: 0x60a5fa,
            opacity: 0.8,
            scatterRange: 300,
            lerpSpeed: 0.08,
            baseLerpSpeed: 0.08,
            autoRotate: true,
            mode: 'normal',
            gcodeColorMode: 'feature',
            showWalls: true,
            showInfill: true,
            showSupport: false,
            backgroundColor: '#050505',
            blendMode: 'solid',
            effectMode: 'none',
            assemblyMode: 'explode',
            
            // Interaction Physics
            interactRadius: 45,
            interactForce: 1.5,
            returnSpring: 0.03,
            damping: 0.92
        };

        // --- State ---
        let scene, camera, renderer, controls, particleSystem, geometry;
        let targetPositions = [], initialPositions = [];
        let velocities = []; 
        let targetColors = []; 
        let isPrinting = false, printProgress = 1.0;
        let time = 0; 
        
        // Interaction
        let raycaster = new THREE.Raycaster();
        let mouse = new THREE.Vector2(-9999, -9999); 
        let interactionTarget = new THREE.Vector3();
        let interactionPlane;
        let isDraggingParticles = false;
        let isCtrlPressed = false;
        
        let rawGCodeData = null; 
        let originalVertices = []; 
        let originalColors = null; 

        // DOM Elements
        const apiKey = ""; 
        const uiElements = {
            loader: document.getElementById('loader'),
            panel: document.getElementById('ui-panel'),
            gcodePanel: document.getElementById('gcode-panel'),
            singleColor: document.getElementById('single-color-ctrl'),
            matList: document.getElementById('materials-list'),
            scatterVal: document.getElementById('scatter-val'),
            scatterCtrl: document.getElementById('scatter-control'),
            countVal: document.getElementById('count-val'),
            printBtn: document.getElementById('btn-print-simulate'),
            colorMode: document.getElementById('color-mode-select'),
            effectMode: document.getElementById('effect-mode'),
            assemblyMode: document.getElementById('assembly-mode'),
            interactRadius: document.getElementById('interact-radius'),
            interactForce: document.getElementById('interact-force'),
            physicsDamping: document.getElementById('physics-damping'),
            tooltip: document.getElementById('interaction-tooltip')
        };

        function init() {
            scene = new THREE.Scene();
            scene.background = new THREE.Color(CONFIG.backgroundColor);
            scene.fog = new THREE.FogExp2(CONFIG.backgroundColor, 0.0025);

            camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 5000);
            camera.position.set(0, 40, 120);

            renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });
            renderer.setSize(window.innerWidth, window.innerHeight);
            renderer.setPixelRatio(window.devicePixelRatio);
            document.getElementById('canvas-container').appendChild(renderer.domElement);

            controls = new OrbitControls(camera, renderer.domElement);
            controls.enableDamping = true;
            controls.autoRotate = CONFIG.autoRotate;
            controls.autoRotateSpeed = 0.8;
            controls.enablePan = true;

            const planeGeo = new THREE.PlaneGeometry(2000, 2000);
            const planeMat = new THREE.MeshBasicMaterial({ visible: false });
            interactionPlane = new THREE.Mesh(planeGeo, planeMat);
            scene.add(interactionPlane);

            createDefaultShape();

            const ambientLight = new THREE.AmbientLight(0xffffff, 0.5);
            scene.add(ambientLight);

            window.addEventListener('resize', onWindowResize);
            
            window.addEventListener('mousemove', onMouseMove);
            window.addEventListener('mousedown', onMouseDown);
            window.addEventListener('mouseup', onMouseUp);
            window.addEventListener('keydown', onKeyDown);
            window.addEventListener('keyup', onKeyUp);

            setupEvents();
            setupAIEvents();
            animate();
            
            setTimeout(() => uiElements.loader.classList.add('hidden-opacity'), 800);
        }

        // --- Interaction Logic ---
        function onMouseMove(event) {
            event.preventDefault();
            mouse.x = (event.clientX / window.innerWidth) * 2 - 1;
            mouse.y = -(event.clientY / window.innerHeight) * 2 + 1;

            if (isDraggingParticles) {
                updateInteractionTarget();
            }
        }

        function onMouseDown(event) {
            if (isCtrlPressed && event.button === 0) { 
                isDraggingParticles = true;
                controls.enabled = false; 
                updateInteractionTarget();
                document.body.style.cursor = 'crosshair';
            }
        }

        function onMouseUp(event) {
            isDraggingParticles = false;
            controls.enabled = true; 
            document.body.style.cursor = 'default';
            interactionTarget.set(9999, 9999, 9999);
        }

        function onKeyDown(event) {
            if (event.key === 'Control') {
                isCtrlPressed = true;
                uiElements.tooltip.style.opacity = 1;
            }
        }

        function onKeyUp(event) {
            if (event.key === 'Control') {
                isCtrlPressed = false;
                if (!isDraggingParticles) {
                    uiElements.tooltip.style.opacity = 0;
                }
            }
        }

        function updateInteractionTarget() {
            raycaster.setFromCamera(mouse, camera);
            interactionPlane.lookAt(camera.position); 
            const intersects = raycaster.intersectObject(interactionPlane);
            if (intersects.length > 0) {
                interactionTarget.copy(intersects[0].point);
            }
        }

        function updateParticle() {
            isPrinting = true;
            printProgress = 0;
            uiElements.scatterCtrl.value = 0;

            if (CONFIG.mode === 'gcode' && rawGCodeData) {
                rebuildGCodeParticles();
            } else if (originalVertices.length > 0) {
                createStandardParticles(originalVertices, originalColors);
            }
        }

        function getMaterialOptions() {
            let blending = THREE.NormalBlending;
            if (CONFIG.blendMode === 'additive' || CONFIG.blendMode === 'subtle' || CONFIG.blendMode === 'plasma') {
                blending = THREE.AdditiveBlending;
            }

            return {
                size: CONFIG.particleSize,
                vertexColors: true,
                transparent: true,
                opacity: (CONFIG.blendMode === 'subtle') ? CONFIG.opacity * 0.3 : CONFIG.opacity,
                sizeAttenuation: true,
                blending: blending,
                depthWrite: !(CONFIG.blendMode === 'additive' || CONFIG.blendMode === 'subtle' || CONFIG.blendMode === 'plasma')
            };
        }

        function getInitialPosition(i, targetX, targetY, targetZ) {
            const mode = CONFIG.assemblyMode;
            if (mode === 'printer') {
                return {
                    x: 0 + (Math.random()-0.5)*20,
                    y: 200 + (Math.random()-0.5)*20,
                    z: 0 + (Math.random()-0.5)*20
                };
            } else if (mode === 'flat') {
                return {
                    x: targetX,
                    y: -100 + (Math.random()-0.5)*10, 
                    z: targetZ
                };
            } else { 
                return {
                    x: (Math.random()-0.5)*CONFIG.scatterRange*2,
                    y: (Math.random()-0.5)*CONFIG.scatterRange*2,
                    z: (Math.random()-0.5)*CONFIG.scatterRange*2
                };
            }
        }

        function createStandardParticles(vertices, colors) {
            if(particleSystem) { scene.remove(particleSystem); geometry.dispose(); }

            const totalPoints = vertices.length / 3;
            const count = Math.min(totalPoints, CONFIG.particleCount);
            
            const pos = new Float32Array(count * 3);
            const col = new Float32Array(count * 3);
            velocities = new Float32Array(count * 3);
            targetPositions = new Float32Array(count * 3);
            initialPositions = new Float32Array(count * 3);
            targetColors = new Float32Array(count * 3);

            const hasImgColors = !!colors;
            const baseColor = CONFIG.color;

            for(let i=0; i<count; i++) {
                let srcIdx = Math.floor(i * (totalPoints / count)) * 3;
                if(srcIdx > vertices.length - 3) srcIdx = vertices.length - 3;

                const tx = vertices[srcIdx], ty = vertices[srcIdx+1], tz = vertices[srcIdx+2];
                
                targetPositions[i*3] = tx;
                targetPositions[i*3+1] = ty;
                targetPositions[i*3+2] = tz;

                if(hasImgColors) {
                    col[i*3] = colors[srcIdx]; col[i*3+1] = colors[srcIdx+1]; col[i*3+2] = colors[srcIdx+2];
                } else {
                    col[i*3] = baseColor.r; col[i*3+1] = baseColor.g; col[i*3+2] = baseColor.b;
                }
                
                targetColors[i*3] = col[i*3];
                targetColors[i*3+1] = col[i*3+1];
                targetColors[i*3+2] = col[i*3+2];

                const initP = getInitialPosition(i, tx, ty, tz);
                initialPositions[i*3] = initP.x;
                initialPositions[i*3+1] = initP.y;
                initialPositions[i*3+2] = initP.z;

                pos[i*3] = initP.x;
                pos[i*3+1] = initP.y;
                pos[i*3+2] = initP.z;
                
                velocities[i*3] = 0; velocities[i*3+1] = 0; velocities[i*3+2] = 0;
            }

            geometry = new THREE.BufferGeometry();
            geometry.setAttribute('position', new THREE.BufferAttribute(pos, 3));
            geometry.setAttribute('color', new THREE.BufferAttribute(col, 3));

            const mat = new THREE.PointsMaterial(getMaterialOptions());

            particleSystem = new THREE.Points(geometry, mat);
            scene.add(particleSystem);
        }

        function rebuildGCodeParticles() {
            if(particleSystem) { scene.remove(particleSystem); geometry.dispose(); }

            const filteredPoints = rawGCodeData.points.filter(p => {
                if (p.feature === 'wall' && !CONFIG.showWalls) return false;
                if (p.feature === 'support' && !CONFIG.showSupport) return false;
                if (p.feature === 'infill' && !CONFIG.showInfill) return false;
                return true;
            });

            const totalPoints = filteredPoints.length;
            if(totalPoints === 0) return;

            const count = Math.min(totalPoints, CONFIG.particleCount);
            
            const pos = new Float32Array(count * 3);
            const col = new Float32Array(count * 3);
            velocities = new Float32Array(count * 3);
            targetPositions = new Float32Array(count * 3);
            initialPositions = new Float32Array(count * 3);
            targetColors = new Float32Array(count * 3);

            const colorWall = new THREE.Color(0x3b82f6); 
            const colorInfill = new THREE.Color(0xf97316); 
            const colorSupport = new THREE.Color(0x94a3b8); 
            const colorSkin = new THREE.Color(0xec4899); 

            const spawnX = 0;
            const spawnY = 200; 
            const spawnZ = 0;

            for(let i=0; i<count; i++) {
                const srcIdx = Math.floor(i * (totalPoints / count));
                const p = filteredPoints[srcIdx];

                targetPositions[i*3] = p.x;
                targetPositions[i*3+1] = p.y;
                targetPositions[i*3+2] = p.z;

                let c = new THREE.Color(0xffffff);
                
                if (CONFIG.gcodeColorMode === 'feature') {
                    if (p.feature === 'wall') c = colorWall;
                    else if (p.feature === 'skin' || p.feature === 'solid') c = colorSkin;
                    else if (p.feature === 'infill') c = colorInfill;
                    else if (p.feature === 'support') c = colorSupport;
                } else {
                    const hex = rawGCodeData.toolColors[p.t] || "#ffffff";
                    c.set(hex);
                }

                col[i*3] = c.r; col[i*3+1] = c.g; col[i*3+2] = c.b;
                targetColors[i*3] = c.r; targetColors[i*3+1] = c.g; targetColors[i*3+2] = c.b;

                // GCode Initial: Top Down Spray
                initialPositions[i*3] = spawnX + (Math.random()-0.5)*20; 
                initialPositions[i*3+1] = spawnY + (Math.random()-0.5)*20;
                initialPositions[i*3+2] = spawnZ + (Math.random()-0.5)*20;

                pos[i*3]=p.x; pos[i*3+1]=p.y; pos[i*3+2]=p.z;
                velocities[i*3] = 0; velocities[i*3+1] = 0; velocities[i*3+2] = 0;
            }

            geometry = new THREE.BufferGeometry();
            geometry.setAttribute('position', new THREE.BufferAttribute(pos, 3));
            geometry.setAttribute('color', new THREE.BufferAttribute(col, 3));

            const mat = new THREE.PointsMaterial(getMaterialOptions());

            particleSystem = new THREE.Points(geometry, mat);
            scene.add(particleSystem);
        }

        function handleGCode(text) {
            const toolColorMap = {};
            const lines = text.split('\n');
            for(let i=0; i<Math.min(lines.length, 2000); i++) {
                const line = lines[i];
                if(/filament_colou?r|color\s*=/i.test(line)) {
                    const matches = line.match(/#[0-9A-Fa-f]{6}/g);
                    if(matches && matches.length > 0) {
                        matches.forEach((color, idx) => {
                            toolColorMap[idx] = color;
                        });
                    }
                }
            }

            const palette = ["#60a5fa", "#f87171", "#4ade80", "#facc15", "#c084fc"];
            
            let x=0, y=0, z=0, e=0;
            let currentTool = 0;
            let currentFeature = 'wall'; 
            let isRelativeE = false;
            let lastE = 0;

            const points = [];
            const detectedTools = new Set([0]);
            
            let minX=Infinity, maxX=-Infinity, minY=Infinity, maxY=-Infinity, minZ=Infinity, maxZ=-Infinity;

            const density = 0.4; 

            for(let line of lines) {
                const l = line.trim().toUpperCase();
                if(l.length === 0) continue;

                if(line.includes(';TYPE:') || line.includes(';FEATURE:')) {
                    const type = line.split(line.includes(';TYPE:') ? ';TYPE:' : ';FEATURE:')[1].trim().toLowerCase();
                    if(type.includes('wall') || type.includes('perimeter') || type.includes('outer') || type.includes('inner')) currentFeature = 'wall';
                    else if(type.includes('support')) currentFeature = 'support';
                    else if(type.includes('solid') || type.includes('top') || type.includes('bottom') || type.includes('skin')) currentFeature = 'skin'; 
                    else if(type.includes('infill')) currentFeature = 'infill'; 
                    else currentFeature = 'wall'; 
                }

                if(l.startsWith('T')) {
                    const t = parseInt(l.substring(1));
                    if(!isNaN(t)) { currentTool = t; detectedTools.add(t); }
                }
                if(l.startsWith('M82')) isRelativeE = false;
                if(l.startsWith('M83')) isRelativeE = true;
                if(l.startsWith('G92')) {
                    const parts = l.split(' ');
                    parts.forEach(p => { if(p.startsWith('E')) lastE = parseFloat(p.substring(1)); });
                }

                if(l.startsWith('G1')) {
                    const parts = l.split(' ');
                    let nx=x, ny=y, nz=z, ne=lastE;
                    let hasMove = false;
                    let hasE = false;

                    parts.forEach(p => {
                        const code = p.charAt(0);
                        const val = parseFloat(p.substring(1));
                        if(code === 'X') { nx = val; hasMove=true; }
                        if(code === 'Y') { ny = val; hasMove=true; }
                        if(code === 'Z') { nz = val; }
                        if(code === 'E') { ne = val; hasE=true; }
                    });

                    let extrusion = 0;
                    if(hasE) {
                        extrusion = isRelativeE ? ne : ne - lastE;
                        lastE = ne;
                    }

                    if(hasMove && extrusion > 0) {
                        const dx = nx - x;
                        const dy = ny - y;
                        const dz = nz - z;
                        const dist = Math.sqrt(dx*dx + dy*dy + dz*dz);

                        if(dist > 0) {
                            const steps = Math.ceil(dist / density);
                            for(let k=1; k<=steps; k++) {
                                const t = k/steps;
                                const px = x + dx*t;
                                const py = y + dy*t;
                                const pz = z + dz*t;
                                
                                points.push({ x: px, y: py, z: pz, t: currentTool, feature: currentFeature });
                                
                                if(px<minX) minX=px; if(px>maxX) maxX=px;
                                if(py<minY) minY=py; if(py>maxY) maxY=py;
                                if(pz<minZ) minZ=pz; if(pz>maxZ) maxZ=pz;
                            }
                        }
                    }
                    x=nx; y=ny; z=nz;
                } else if (l.startsWith('G0')) {
                    const parts = l.split(' ');
                    parts.forEach(p => {
                        const code = p.charAt(0);
                        const val = parseFloat(p.substring(1));
                        if(code === 'X') x=val; if(code === 'Y') y=val; if(code === 'Z') z=val;
                    });
                }
            }

            if(points.length === 0) { alert("无效 GCode (无挤出路径)"); return; }

            const cx = (minX+maxX)/2;
            const cy = (minY+maxY)/2;
            const midZ = (minZ+maxZ)/2;
            const scale = 100 / Math.max(maxX-minX, maxY-minY, maxZ-minZ);

            points.forEach(p => {
                const ox = p.x, oy = p.y, oz = p.z;
                p.x = (ox - cx) * scale;
                p.y = (oz - minZ) * scale - 20; // Z becomes Y (Up)
                p.z = (oy - cy) * scale;        // Y becomes Z (Depth)
            });

            rawGCodeData = { points: points, toolColors: {} };

            // Determine Start Mode
            const hasHeaderColors = Object.keys(toolColorMap).length > 0;
            CONFIG.gcodeColorMode = hasHeaderColors ? 'material' : 'feature';
            uiElements.colorMode.value = CONFIG.gcodeColorMode;

            uiElements.matList.innerHTML = '';
            Array.from(detectedTools).sort().forEach((t, i) => {
                const defColor = toolColorMap[t] || palette[i%palette.length];
                rawGCodeData.toolColors[t] = defColor;
                const div = document.createElement('div');
                div.className = "flex justify-between text-[10px] items-center text-slate-400";
                div.innerHTML = `<span>材料 T${t}</span><span class="w-3 h-3 rounded-full" style="background:${defColor}"></span>`;
                uiElements.matList.appendChild(div);
            });
            uiElements.matList.classList.remove('hidden');

            CONFIG.mode = 'gcode';
            uiElements.gcodePanel.classList.remove('hidden');
            uiElements.singleColor.classList.add('hidden');
            uiElements.scatterVal.textContent = '100%';
            uiElements.scatterCtrl.value = 1;
            printProgress = 1;

            updateParticle();
        }

        function handleSTL(buffer) {
            const geometry = new STLLoader().parse(buffer);
            geometry.center();
            geometry.computeBoundingBox();
            const bbox = geometry.boundingBox;
            const maxDim = Math.max(bbox.max.x-bbox.min.x, bbox.max.y-bbox.min.y, bbox.max.z-bbox.min.z);
            const scale = 100/maxDim;
            
            const arr = geometry.attributes.position.array;
            const vertices = [];
            for(let i=0; i<arr.length; i++) vertices.push(arr[i]*scale);

            // FIX: Reset State for STL and force 100% assembly
            CONFIG.mode = 'normal';
            rawGCodeData = null; // Clear GCode data
            originalVertices = vertices;
            originalColors = null;
            
            uiElements.gcodePanel.classList.add('hidden');
            uiElements.singleColor.classList.remove('hidden');
            
            // Force reset slider
            uiElements.scatterCtrl.value = 1;
            uiElements.scatterVal.textContent = '100%';
            
            // Explicitly reset particle system
            updateParticle();
        }

        // --- Animations ---
        function animate() {
            requestAnimationFrame(animate);
            controls.update();
            time += 0.01;

            if(!particleSystem || !geometry) { renderer.render(scene, camera); return; }

            const pos = geometry.attributes.position.array;
            const colors = geometry.attributes.color.array;
            
            // Auto Animation Logic
            if(isPrinting) {
                printProgress += 0.005 * (CONFIG.lerpSpeed * 10);
                if(printProgress > 1) { 
                    printProgress = 1; 
                    isPrinting=false; 
                    uiElements.printBtn.innerHTML='播放打印过程'; 
                }
                uiElements.scatterCtrl.value = printProgress;
                uiElements.scatterVal.textContent = Math.round(printProgress*100)+'%';
            } else {
                printProgress = parseFloat(uiElements.scatterCtrl.value); 
            }

            const activeCount = CONFIG.mode === 'gcode' ? Math.floor((pos.length/3) * printProgress) : pos.length/3;
            // Normal mode always has full count active, interpolation handles position
            const lerpFactor = printProgress; 

            for(let i=0; i<pos.length/3; i++) {
                let tx, ty, tz;
                
                // Determine Target
                tx = targetPositions[i*3];
                ty = targetPositions[i*3+1];
                tz = targetPositions[i*3+2];

                // Apply Effects (Presets) to Target
                if (CONFIG.effectMode === 'breathe') {
                    const s = 1.0 + Math.sin(time * 2.0) * 0.05;
                    tx *= s; ty *= s; tz *= s;
                } else if (CONFIG.effectMode === 'float') {
                    ty += Math.sin(time + tx * 0.05) * 2.0;
                } else if (CONFIG.effectMode === 'vortex') {
                    const angle = time * 0.5 + Math.sqrt(tx*tx + tz*tz) * 0.05;
                    const r = Math.sqrt(tx*tx + tz*tz);
                    tx = Math.cos(angle) * r;
                    tz = Math.sin(angle) * r;
                } else if (CONFIG.effectMode === 'glitch') {
                    if (Math.random() > 0.98) {
                        tx += (Math.random() - 0.5) * 5;
                        ty += (Math.random() - 0.5) * 5;
                        tz += (Math.random() - 0.5) * 5;
                    }
                } else if (CONFIG.effectMode === 'scanline') {
                    const scanY = Math.sin(time * 1.5) * 50;
                    if(Math.abs(ty - scanY) < 5) {
                        tx *= 1.05; tz *= 1.05;
                    }
                } else if (CONFIG.effectMode === 'noise') {
                    tx += (Math.random() - 0.5) * 0.5;
                    ty += (Math.random() - 0.5) * 0.5;
                    tz += (Math.random() - 0.5) * 0.5;
                }

                // Determine Mix Position
                let dx, dy, dz;
                
                if (CONFIG.mode === 'gcode') {
                    if(i < activeCount) {
                        dx = tx - pos[i*3];
                        dy = ty - pos[i*3+1];
                        dz = tz - pos[i*3+2];
                    } else {
                        const ix = initialPositions[i*3];
                        const iy = initialPositions[i*3+1];
                        const iz = initialPositions[i*3+2];
                        dx = ix - pos[i*3];
                        dy = iy - pos[i*3+1];
                        dz = iz - pos[i*3+2];
                    }
                } else {
                    const ix = initialPositions[i*3];
                    const iy = initialPositions[i*3+1];
                    const iz = initialPositions[i*3+2];
                    
                    const cx = ix + (tx-ix)*lerpFactor;
                    const cy = iy + (ty-iy)*lerpFactor;
                    const cz = iz + (tz-iz)*lerpFactor;
                    
                    dx = cx - pos[i*3];
                    dy = cy - pos[i*3+1];
                    dz = cz - pos[i*3+2];
                }

                // VAM Style Soft Physics Logic
                let isPulled = false;
                if (isDraggingParticles && interactionTarget.x !== 9999) {
                    const distSq = (pos[i*3] - interactionTarget.x)**2 + (pos[i*3+1] - interactionTarget.y)**2 + (pos[i*3+2] - interactionTarget.z)**2;
                    const radiusSq = CONFIG.interactRadius * CONFIG.interactRadius;
                    
                    if (distSq < radiusSq) {
                        // Gaussian Falloff
                        const falloff = Math.exp(-distSq / (radiusSq * 0.5)); 
                        
                        // Force
                        const forceX = (interactionTarget.x - pos[i*3]) * CONFIG.interactForce * falloff;
                        const forceY = (interactionTarget.y - pos[i*3+1]) * CONFIG.interactForce * falloff;
                        const forceZ = (interactionTarget.z - pos[i*3+2]) * CONFIG.interactForce * falloff;
                        
                        velocities[i*3] += forceX * 0.1;
                        velocities[i*3+1] += forceY * 0.1;
                        velocities[i*3+2] += forceZ * 0.1;
                        
                        // Color Highlight on Pull
                        if (falloff > 0.2) {
                             isPulled = true;
                             // Invert color logic
                             colors[i*3] = 1.0 - targetColors[i*3];
                             colors[i*3+1] = 1.0 - targetColors[i*3+1];
                             colors[i*3+2] = 1.0 - targetColors[i*3+2];
                             
                             // Add extra glow for visibility
                             if(CONFIG.blendMode === 'additive' || CONFIG.blendMode === 'plasma') {
                                 colors[i*3] += 0.5; colors[i*3+1] += 0.5; colors[i*3+2] += 0.5;
                             }
                        }
                    }
                }

                // Return Spring
                const springX = dx * CONFIG.returnSpring;
                const springY = dy * CONFIG.returnSpring;
                const springZ = dz * CONFIG.returnSpring;
                
                velocities[i*3] += springX;
                velocities[i*3+1] += springY;
                velocities[i*3+2] += springZ;
                
                // Damping
                velocities[i*3] *= CONFIG.damping;
                velocities[i*3+1] *= CONFIG.damping;
                velocities[i*3+2] *= CONFIG.damping;
                
                // Update Pos
                pos[i*3] += velocities[i*3];
                pos[i*3+1] += velocities[i*3+1];
                pos[i*3+2] += velocities[i*3+2];

                // Color Updates
                if (CONFIG.blendMode === 'plasma') {
                    // Time-based cycling for Plasma mode
                    const hue = (time * 0.1 + (i * 0.0001)) % 1;
                    const color = new THREE.Color().setHSL(hue, 0.8, 0.5);
                    if (!isPulled) {
                        colors[i*3] += (color.r - colors[i*3]) * 0.05;
                        colors[i*3+1] += (color.g - colors[i*3+1]) * 0.05;
                        colors[i*3+2] += (color.b - colors[i*3+2]) * 0.05;
                    }
                } else if(!isDraggingParticles && !isPrinting && !isPulled) {
                    // Return to original color
                     colors[i*3] += (targetColors[i*3] - colors[i*3]) * 0.05;
                     colors[i*3+1] += (targetColors[i*3+1] - colors[i*3+1]) * 0.05;
                     colors[i*3+2] += (targetColors[i*3+2] - colors[i*3+2]) * 0.05;
                }

                // Print Hot Tip
                if(CONFIG.mode === 'gcode' && isPrinting) {
                    const distSq = dx*dx + dy*dy + dz*dz;
                    if(distSq > 5.0) {
                        colors[i*3] = 1.0; colors[i*3+1] = 1.0; colors[i*3+2] = 1.0;
                    }
                }
            }

            if(CONFIG.mode === 'gcode') {
                geometry.setDrawRange(0, Infinity);
            } else {
                geometry.setDrawRange(0, Infinity);
            }

            geometry.attributes.position.needsUpdate = true;
            geometry.attributes.color.needsUpdate = true; 
            renderer.render(scene, camera);
        }

        function handleImage(imageSrc) {
            const img = new Image();
            img.crossOrigin = "Anonymous"; 
            img.src = imageSrc;
            img.onload = () => {
                const canvas = document.createElement('canvas');
                const ctx = canvas.getContext('2d');
                const maxSize = 1500; 
                let width = img.width;
                let height = img.height;
                if (width > height) { if (width > maxSize) { height *= maxSize / width; width = maxSize; } } else { if (height > maxSize) { width *= maxSize / height; height = maxSize; } }
                canvas.width = width; canvas.height = height;
                ctx.drawImage(img, 0, 0, width, height);
                const imgData = ctx.getImageData(0, 0, width, height);
                const data = imgData.data; 
                const vertices = [];
                const colors = []; 
                const scaleFactor = 100 / Math.max(width, height);

                for (let y = 0; y < height; y++) {
                    for (let x = 0; x < width; x++) {
                        const index = (y * width + x) * 4;
                        const r = data[index], g = data[index + 1], b = data[index + 2], a = data[index + 3];
                        if (a > 64 && (r+g+b)/3 > 10) {
                            vertices.push((x - width / 2) * scaleFactor); 
                            vertices.push(-(y - height / 2) * scaleFactor); 
                            vertices.push((Math.random() - 0.5) * 5); 
                            colors.push(r / 255); colors.push(g / 255); colors.push(b / 255);
                        }
                    }
                }
                
                CONFIG.mode = 'normal';
                originalVertices = vertices;
                originalColors = colors;
                
                uiElements.gcodePanel.classList.add('hidden');
                uiElements.singleColor.classList.remove('hidden');
                
                updateParticle();
            };
        }
